(D) Find if number is prime or not
  Pseudocode:
    1. Start
    2. Input n
    3. let div = 2
    4. while div < n do
        if n%div == 0 do
          print "Not prime"
          Exit
        else 
          div = div + 1      
    5. print "prime"
    6. Exit

(E) Find sum of first n natural number
  Pseudocode:
    1. Start
    2. Input n
    3. let sum = 0, val = 1
    4. while val < n do
        sum = sum + val
        val = val + 1      
    5. print sum
    6. Exit
